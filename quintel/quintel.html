Language 
A - copy a board to focus
#1100/0101 - copy a constant to focus
o3x5 - an empty board with height 3, width 5
>B - copy focus to a board
&(B), |(B), ^(B) - bitwise and, or, xor with another board
&&[lrtb], ||[lrtb], ^^[lrtb] - bitwise and, or, xor focus for each row/column
~ - bitwise not
=(B) - test equality against another board
a[lrtb](B) - append board in direction
t[lrtb][#] - take rows/columns from direction
d[lrtb][#] - drop rows/columns from direction
c[lrtb][#] - copy board in direction
x[lrtb][#] - extend board in direction with zeros
s[lrtb][#] - shift board in direction, fill with zeros
r[lrtb][#] - roll board in direction
mh, mv, md, mz - mirror horizontally, vertically, main-diagonally, secondary-diagonally
mc, ma, mu - rotate clockwise, rotate anticlockwise, rotate upside-down

[lrtb] - direction, one of l, r, t, and b
[#] - number
